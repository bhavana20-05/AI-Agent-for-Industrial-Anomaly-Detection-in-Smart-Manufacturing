# Your FeatCAE model
class FeatCAE(nn.Module):
    """Autoencoder."""

    def __init__(self, in_channels=1000, latent_dim=50, is_bn=True):
        super(FeatCAE, self).__init__()

        layers = []
        layers += [nn.Conv2d(in_channels, (in_channels + 2 * latent_dim) // 2, kernel_size=1, stride=1, padding=0)]
        if is_bn:
            layers += [nn.BatchNorm2d(num_features=(in_channels + 2 * latent_dim) // 2)]
        layers += [nn.ReLU()]
        layers += [nn.Conv2d((in_channels + 2 * latent_dim) // 2, 2 * latent_dim, kernel_size=1, stride=1, padding=0)]
        if is_bn:
            layers += [nn.BatchNorm2d(num_features=2 * latent_dim)]
        layers += [nn.ReLU()]
        layers += [nn.Conv2d(2 * latent_dim, latent_dim, kernel_size=1, stride=1, padding=0)]

        self.encoder = nn.Sequential(*layers)

        # if 1x1 conv to reconstruct the rgb values, we try to learn a linear combination
        # of the features for rgb
        layers = []
        layers += [nn.Conv2d(latent_dim, 2 * latent_dim, kernel_size=1, stride=1, padding=0)]
        if is_bn:
            layers += [nn.BatchNorm2d(num_features=2 * latent_dim)]
        layers += [nn.ReLU()]
        layers += [nn.Conv2d(2 * latent_dim, (in_channels + 2 * latent_dim) // 2, kernel_size=1, stride=1, padding=0)]
        if is_bn:
            layers += [nn.BatchNorm2d(num_features=(in_channels + 2 * latent_dim) // 2)]
        layers += [nn.ReLU()]
        layers += [nn.Conv2d((in_channels + 2 * latent_dim) // 2, in_channels, kernel_size=1, stride=1, padding=0)]
        # layers += [nn.ReLU()]

        self.decoder = nn.Sequential(*layers)

    def forward(self, x):
        x = self.encoder(x)
        x = self.decoder(x)
        return x

# Feature extractor using ResNet50 (to match your 1000 channels)
class ResNetFeatureExtractor(nn.Module):
    def __init__(self):
        super(ResNetFeatureExtractor, self).__init__()
        from torchvision.models import resnet50, ResNet50_Weights
        
        # Load pre-trained ResNet50
        self.resnet = resnet50(weights=ResNet50_Weights.DEFAULT)
        
        # Remove the final classification layer
        self.feature_extractor = nn.Sequential(*list(self.resnet.children())[:-2])
        
    def forward(self, x):
        features = self.feature_extractor(x)
        return features

# Simple feature extractor (alternative if ResNet50 is too heavy)
class SimpleFeatureExtractor(nn.Module):
    def __init__(self, output_channels=1000):
        super(SimpleFeatureExtractor, self).__init__()
        self.features = nn.Sequential(
            nn.Conv2d(3, 64, 3, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Conv2d(64, 128, 3, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Conv2d(128, 256, 3, padding=1),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Conv2d(256, 512, 3, padding=1),
            nn.ReLU(),
            nn.AdaptiveAvgPool2d((7, 7)),
            nn.Conv2d(512, output_channels, 1)  # Project to desired channels
        )
    
    def forward(self, x):
        return self.features(x)

# Set device
device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
print(f"Using device: {device}")

# Initialize models - Using 1000 channels to match ResNet features
in_channels = 1000
latent_dim = 100  # You can adjust this

model = FeatCAE(in_channels=in_channels, latent_dim=latent_dim, is_bn=True).to(device)
# Choose one feature extractor:
backbone = SimpleFeatureExtractor(output_channels=in_channels).to(device)
# Or use ResNet: backbone = ResNetFeatureExtractor().to(device)

# Print model information
print(f"FeatCAE parameters: {sum(p.numel() for p in model.parameters()):,}")
print(f"Feature extractor parameters: {sum(p.numel() for p in backbone.parameters()):,}")

# Define loss and optimizer
criterion = nn.MSELoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)

print("Models initialized successfully!")

# Load your dataset
try:
    transform = transforms.Compose([
        transforms.Resize((224, 224)),
        transforms.ToTensor()
    ])
    
    train_image_path = Path('carpet/train')
    good_dataset = ImageFolder(root=train_image_path, transform=transform)
    
    # Split dataset
    train_size = int(0.8 * len(good_dataset))
    val_size = len(good_dataset) - train_size
    train_dataset, val_dataset = torch.utils.data.random_split(good_dataset, [train_size, val_size])
    
    # Data loaders
    train_loader = DataLoader(train_dataset, batch_size=8, shuffle=True)
    val_loader = DataLoader(val_dataset, batch_size=8, shuffle=False)
    
    print(f"Training samples: {len(train_dataset)}")
    print(f"Validation samples: {len(val_dataset)}")
    
except FileNotFoundError as e:
    print(f"Dataset error: {e}")
    print("Please check if 'carpet/train' directory exists with proper structure")
    exit()

# Training function - CORRECTED PRINT STATEMENT
def train_model(model, backbone, train_loader, epochs=10):
    model.train()
    backbone.eval()  # Freeze backbone
    
    train_losses = []
    
    for epoch in range(epochs):
        epoch_loss = 0.0
        batch_count = 0
        
        for batch_idx, (images, _) in enumerate(train_loader):
            images = images.to(device)
            
            # Extract features (no gradients for backbone)
            with torch.no_grad():
                features = backbone(images)
            
            # Forward pass
            reconstructed = model(features)
            loss = criterion(reconstructed, features)
            
            # Backward pass
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            
            epoch_loss += loss.item()
            batch_count += 1
            
            if batch_idx % 10 == 0:
                print(f'Epoch [{epoch+1}/{epochs}], Batch [{batch_idx}/{len(train_loader)}], Loss: {loss.item():.6f}')
        
        avg_loss = epoch_loss / batch_count
        train_losses.append(avg_loss)
        print(f'Epoch [{epoch+1}/{epochs}], Average Loss: {avg_loss:.6f}')  # FIXED: Removed extra dot
    
    return train_losses

# Train the model
print("Starting training...")
train_losses = train_model(model, backbone, train_loader, epochs=5)

# Save the trained model
torch.save(model.state_dict(), 'autoencoder_with_resnet_deep_features.pth')
