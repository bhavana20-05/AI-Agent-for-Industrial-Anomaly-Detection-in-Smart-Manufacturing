from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy
from flask_cors import CORS
from datetime import datetime
import os

app = Flask(__name__)
CORS(app)

# PostgreSQL Config
app.config['SQLALCHEMY_DATABASE_URI'] = 'postgresql://postgres:password@localhost:5432/carpet_ai'
app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False
app.config['UPLOAD_FOLDER'] = 'uploads'

db = SQLAlchemy(app)

# Model
class Detection(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    filename = db.Column(db.String(200))
    result = db.Column(db.String(500))
    date = db.Column(db.DateTime, default=datetime.utcnow)

# Create Tables
with app.app_context():
    db.create_all()

# Routes
@app.route("/analyze", methods=["POST"])
def analyze():
    files = request.files.getlist("images")
    responses = []

    for file in files:
        filepath = os.path.join(app.config['UPLOAD_FOLDER'], file.filename)
        file.save(filepath)

        result = "No defects detected"

        record = Detection(filename=file.filename, result=result)
        db.session.add(record)

        responses.append({
            "filename": file.filename,
            "result": result
        })

    db.session.commit()

    return jsonify(responses)

@app.route("/history", methods=["GET"])
def history():
    records = Detection.query.order_by(Detection.date.desc()).all()

    return jsonify([
        {
            "id": r.id,
            "filename": r.filename,
            "result": r.result,
            "date": r.date
        } for r in records
    ])

if __name__ == "__main__":
    app.run(debug=True)
